from typing import List, Iterator, TypeVar

def reservoir_sampling(stream: Iterator[T], k: int) -> List[T]:

import random
from typing import List, Iterator, TypeVar
import math

T = TypeVar('T')

class OptimizedReservoirSampler:
    def __init__(self, k: int):
        self.k = k
        self.reservoir: List[T] = []
        self.count = 0
        
    def _skip_count(self) -> int:
        """–†–æ–∑—Ä–∞—Ö–æ–≤—É—î –∫—ñ–ª—å–∫—ñ—Å—Ç—å –µ–ª–µ–º–µ–Ω—Ç—ñ–≤, —è–∫—ñ –º–æ–∂–Ω–∞ –ø—Ä–æ–ø—É—Å—Ç–∏—Ç–∏."""
        gap = random.random()
        return math.floor(
            math.log(gap) / 
            math.log(1 - self.k/max(self.count, self.k))
        )
    
    def sample(self, stream: Iterator[T]) -> List[T]:
        """
        –û–ø—Ç–∏–º—ñ–∑–æ–≤–∞–Ω–∞ –≤–µ—Ä—Å—ñ—è —Ä–µ–∑–µ—Ä–≤—É–∞—Ä–Ω–æ—ó –≤–∏–±—ñ—Ä–∫–∏.

            stream: –Ü—Ç–µ—Ä–∞—Ç–æ—Ä –µ–ª–µ–º–µ–Ω—Ç—ñ–≤

            List[T]: –í–∏–ø–∞–¥–∫–æ–≤–∞ –≤–∏–±—ñ—Ä–∫–∞ —Ä–æ–∑–º—ñ—Ä—É k
        """
        stream = iter(stream)

        for _ in range(self.k):
            try:
                self.reservoir.append(next(stream))
                self.count += 1
            except StopIteration:
                return self.reservoir
        
        # –û–±—Ä–æ–±–ª—è—î–º–æ —Ä–µ—à—Ç—É –ø–æ—Ç–æ–∫—É –∑ –æ–ø—Ç–∏–º—ñ–∑–∞—Ü—ñ—î—é
        while True:
            try:
                skip = self._skip_count()
                
                # –ü—Ä–æ–ø—É—Å–∫–∞—î–º–æ –µ–ª–µ–º–µ–Ω—Ç–∏

                    next(stream)
                    self.count += 1
                
                # –ë–µ—Ä–µ–º–æ –Ω–∞—Å—Ç—É–ø–Ω–∏–π –µ–ª–µ–º–µ–Ω—Ç
                item = next(stream)
                self.count += 1

                j = random.randint(0, self.k - 1)
                self.reservoir[j] = item
                
            except StopIteration:
                break
        
        return self.reservoir
